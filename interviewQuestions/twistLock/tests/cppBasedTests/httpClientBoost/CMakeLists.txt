# ======================================================================================================================================
# general project settings
# ======================================================================================================================================
cmake_minimum_required(VERSION 3.5)

project(cppComponentTests)

message(STATUS "processing CMakeLists of project:${CMAKE_PROJECT_NAME} within folder:${CMAKE_CURRENT_LIST_DIR}")

# ======================================================================================================================================
# include directories
# ======================================================================================================================================


# ======================================================================================================================================
# group all source files of the library
# ======================================================================================================================================
file(GLOB_RECURSE TS_CPP_TESTS_SRC
    "${CMAKE_CURRENT_LIST_DIR}/src/*.c"
    "${CMAKE_CURRENT_LIST_DIR}/src/*.cpp"
    "${CMAKE_CURRENT_LIST_DIR}/include/*.h") 

file(GLOB_RECURSE TS_CPP_TESTS_HEADERS
    "${CMAKE_CURRENT_LIST_DIR}/*.h")

# ======================================================================================================================================
# create the library
# ======================================================================================================================================

add_executable(httpClientCpp ${TS_CPP_TESTS_SRC} ${TS_CPP_TESTS_HEADERS})

# ======================================================================================================================================
# link the library with all other needed libraries
# ======================================================================================================================================

target_link_libraries(httpClientCpp pthread boost_system)

# ======================================================================================================================================
# general project settings
# ======================================================================================================================================
cmake_minimum_required(VERSION 3.5)

project(common)

message(STATUS "processing CMakeLists of project:${CMAKE_PROJECT_NAME} within folder:${CMAKE_CURRENT_LIST_DIR}")

# ======================================================================================================================================
# include directories
# ======================================================================================================================================

include_directories(${CMAKE_CURRENT_LIST_DIR}/include)

# ======================================================================================================================================
# group all source files of the library
# ======================================================================================================================================
file(GLOB_RECURSE TS_COMMON_SRC
    "${CMAKE_CURRENT_LIST_DIR}/src/*.c"
    "${CMAKE_CURRENT_LIST_DIR}/src/*.cpp") 

file(GLOB_RECURSE TS_COMMON_HEADERS
    "${CMAKE_CURRENT_LIST_DIR}/include/*.h")

# ======================================================================================================================================
# create the library
# ======================================================================================================================================
add_library(common SHARED ${TS_COMMON_SRC} ${TS_COMMON_HEADERS})
